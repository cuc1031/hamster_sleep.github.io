const data = getData();
const box = document.getElementById("historyBox");

let html = "";

// ⭐ 先排序日期
const dates = Object.keys(records).sort().reverse();

// ⭐ 計算每一天的「累積總額」
let runningTotal = 0;
const totals = {};

Object.keys(records).sort().forEach(d=>{
runningTotal += (records[d].delta || 0);
totals[d] = runningTotal;
});

// ⭐ 渲染
dates.forEach(d=>{
const r = records[d];

let resultText = r.result || "";
if(!resultText && r.reward){
resultText = "每周獎勵";
}
if(!resultText){
resultText = "未完成";
}

html += `


${d} ｜ ${resultText} ｜ ${r.delta || 0} ｜ 總額 ${totals[d]}


`;
});

box.innerHTML = html || "尚無紀錄";

}
