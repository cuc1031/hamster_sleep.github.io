function checkIn(type) { initWeek(); const data = getData(); const today = getTodayKey(); if (!data.records) data.records = {}; if (!data.records[today]) data.records[today] = {}; const record = data.records[today]; // 如果今天已經判定完成就不能再打卡 if (record.evaluated) { alert("今天已經完成紀錄，不能再打卡！"); return; } // 睡覺卡只能打一次 if (type === 'sleep') { if (record.sleep) { alert("今天已經打過睡覺卡了！"); return; } record.sleep = new Date().toISOString(); } // 起床卡限制 if (type === 'wake') { // 必須先睡覺 if (!record.sleep) { alert("請先打睡覺卡！"); return; } // 起床卡只能一次 if (record.wake) { alert("今天已經打過起床卡了！"); return; } record.wake = new Date().toISOString(); } evaluateDay(today, data); saveData(data); render(); }
